CREATE TABLE "PAYMENT" 
   (	"PAYMENT_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"PAYMENT_DATE" DATE, 
	"PAYMENT_TYPE" VARCHAR2(100), 
	"PAYMENT_AMOUNT" NUMBER(*,0), 
	"CUSTOMER_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"ORDER_PRODUCT_ID" NUMBER(*,0) NOT NULL ENABLE, 
	 CONSTRAINT "PK_PAYMENT_ID" PRIMARY KEY ("PAYMENT_ID")
  USING INDEX  ENABLE
   ) ;

  ALTER TABLE "PAYMENT" ADD CONSTRAINT "FK_PAYMENT_CUSTOMER_ID" FOREIGN KEY ("CUSTOMER_ID")
	  REFERENCES "CUSTOMER" ("CUSTOMER_ID") ON DELETE CASCADE ENABLE;

  CREATE OR REPLACE EDITIONABLE TRIGGER "PAYMENT_TRG" 
BEFORE INSERT ON PAYMENT 
FOR EACH ROW 
BEGIN  
  IF :NEW.payment_id IS NULL THEN  
  SELECT seq_payment.NEXTVAL INTO :NEW.payment_id FROM SYS.DUAL; 
  END IF; 
END;
/
ALTER TRIGGER "PAYMENT_TRG" ENABLE;
