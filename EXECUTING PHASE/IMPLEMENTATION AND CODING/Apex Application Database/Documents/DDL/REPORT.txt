CREATE TABLE "REPORT" 
   (	"REPORT_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"REPORT_TYPE" VARCHAR2(100), 
	"REPORT_NAME" VARCHAR2(100), 
	"REPORT_DATE" DATE, 
	"PRODUCT_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"USER_ID" NUMBER(*,0), 
	 CONSTRAINT "PK_REPORT_ID" PRIMARY KEY ("REPORT_ID")
  USING INDEX  ENABLE
   ) ;

  ALTER TABLE "REPORT" ADD CONSTRAINT "FK_REPORT_PRODUCT_ID" FOREIGN KEY ("PRODUCT_ID")
	  REFERENCES "PRODUCT" ("PRODUCT_ID") ON DELETE CASCADE ENABLE;
  ALTER TABLE "REPORT" ADD CONSTRAINT "FK_REPORT_USER_ID" FOREIGN KEY ("USER_ID")
	  REFERENCES "CLECK_USER" ("USER_ID") ON DELETE CASCADE ENABLE;

  CREATE OR REPLACE EDITIONABLE TRIGGER "REPORT_TRG" 
BEFORE INSERT ON REPORT 
FOR EACH ROW 
BEGIN  
  IF :NEW.report_id IS NULL THEN  
  SELECT seq_report.NEXTVAL INTO :NEW.report_id FROM SYS.DUAL; 
  END IF; 
END;
/
ALTER TRIGGER "REPORT_TRG" ENABLE;
