CREATE TABLE "TRADER" 
   (	"TRADER_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"TRADER_TYPE" VARCHAR2(100), 
	"SHOP_NAME" VARCHAR2(100), 
	"VERIFICATION_STATUS" NUMBER(1,0), 
	"USER_ID" NUMBER(*,0), 
	"PROFILE_PICTURE" VARCHAR2(255), 
	"VERIFICATION_CODE" NUMBER, 
	"VERFIED_ADMIN" NUMBER, 
	"VERIFICATION_SEND" NUMBER, 
	 CONSTRAINT "PK_TRADER_ID" PRIMARY KEY ("TRADER_ID")
  USING INDEX  ENABLE
   ) ;

  ALTER TABLE "TRADER" ADD CONSTRAINT "FK_TRADER_ID" FOREIGN KEY ("USER_ID")
	  REFERENCES "CLECK_USER" ("USER_ID") ON DELETE CASCADE ENABLE;

  CREATE OR REPLACE EDITIONABLE TRIGGER "TRADER_TRG" 
BEFORE INSERT ON TRADER 
FOR EACH ROW 
BEGIN  
  IF :NEW.trader_id IS NULL THEN  
  SELECT seq_trader.NEXTVAL INTO :NEW.trader_id FROM SYS.DUAL; 
  END IF; 
END;
/
ALTER TRIGGER "TRADER_TRG" ENABLE;
