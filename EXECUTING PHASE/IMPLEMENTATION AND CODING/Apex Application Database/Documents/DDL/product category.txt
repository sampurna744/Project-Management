CREATE TABLE "PRODUCT_CATEGORY" 
   (	"CATEGORY_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"CATEGORY_TYPE" VARCHAR2(100), 
	"CATEGORY_IMAGE" VARCHAR2(255), 
	 CONSTRAINT "PK_CATEGORY_ID" PRIMARY KEY ("CATEGORY_ID")
  USING INDEX  ENABLE
   ) ;

  CREATE OR REPLACE EDITIONABLE TRIGGER "CATEGORY_TRG" 
BEFORE INSERT ON PRODUCT_CATEGORY 
FOR EACH ROW 
BEGIN  
  IF :NEW.category_id IS NULL THEN  
  SELECT seq_category.NEXTVAL INTO :NEW.category_id FROM SYS.DUAL; 
  END IF; 
END;
/
ALTER TRIGGER "CATEGORY_TRG" ENABLE;
