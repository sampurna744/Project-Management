CREATE TABLE "SHOP" 
   (	"SHOP_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"SHOP_NAME" VARCHAR2(100), 
	"SHOP_DESCRIPTION" VARCHAR2(500), 
	"USER_ID" NUMBER(*,0), 
	"VERIFIED_SHOP" NUMBER(1,0), 
	"SHOP_PROFILE" VARCHAR2(255), 
	"SHOP_CATEGORY_ID" NUMBER, 
	"REGISTRATION_NO" NUMBER, 
	 CONSTRAINT "PK_SHOP_ID" PRIMARY KEY ("SHOP_ID")
  USING INDEX  ENABLE
   ) ;

  ALTER TABLE "SHOP" ADD CONSTRAINT "FK_SHOP_USER_ID" FOREIGN KEY ("USER_ID")
	  REFERENCES "CLECK_USER" ("USER_ID") ON DELETE CASCADE ENABLE;

  CREATE OR REPLACE EDITIONABLE TRIGGER "SHOP_TRG" 
BEFORE INSERT ON SHOP 
FOR EACH ROW 
BEGIN  
  IF :NEW.shop_id IS NULL THEN  
  SELECT seq_shop.NEXTVAL INTO :NEW.shop_id FROM SYS.DUAL; 
  END IF; 
END;
/
ALTER TRIGGER "SHOP_TRG" ENABLE;
