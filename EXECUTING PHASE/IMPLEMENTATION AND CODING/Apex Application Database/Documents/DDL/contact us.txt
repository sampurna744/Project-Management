CREATE TABLE "CONTACTUS" 
   (	"QUERY_ID" NUMBER(10,0) NOT NULL ENABLE, 
	"FIRST_NAME" VARCHAR2(100), 
	"LAST_NAME" VARCHAR2(100), 
	"EMAIL" VARCHAR2(100), 
	"CONTACT_NO" VARCHAR2(20), 
	"SUBJECT" VARCHAR2(255), 
	"MESSAGE" VARCHAR2(1000), 
	"VERIFIED_STATUS" NUMBER(1,0) DEFAULT 0, 
	 CONSTRAINT "PK_QUERY_ID" PRIMARY KEY ("QUERY_ID")
  USING INDEX  ENABLE
   ) ;

  CREATE OR REPLACE EDITIONABLE TRIGGER "CONTACT_ID_TRIGGER" 
BEFORE INSERT ON CONTACTUS 
FOR EACH ROW 
BEGIN 
  IF :NEW.query_id IS NULL THEN 
    SELECT contact_id_seq.NEXTVAL INTO :NEW.query_id FROM DUAL; 
  END IF; 
END;
/
ALTER TRIGGER "CONTACT_ID_TRIGGER" ENABLE;
