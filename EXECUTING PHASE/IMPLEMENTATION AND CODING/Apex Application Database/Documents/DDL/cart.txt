CREATE TABLE "CART" 
   (	"CART_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"CUSTOMER_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"ORDER_PRODUCT_ID" NUMBER(*,0) NOT NULL ENABLE, 
	 CONSTRAINT "PK_CART_ID" PRIMARY KEY ("CART_ID")
  USING INDEX  ENABLE
   ) ;

  ALTER TABLE "CART" ADD CONSTRAINT "FK_CUSTOMER_CART_ID" FOREIGN KEY ("CUSTOMER_ID")
	  REFERENCES "CUSTOMER" ("CUSTOMER_ID") ON DELETE CASCADE ENABLE;

  CREATE OR REPLACE EDITIONABLE TRIGGER "CART_TRG" 
BEFORE INSERT ON CART 
FOR EACH ROW 
BEGIN  
  IF :NEW.cart_id IS NULL THEN  
  SELECT seq_cart.NEXTVAL INTO :NEW.cart_id FROM SYS.DUAL; 
  END IF; 
END;
/
ALTER TRIGGER "CART_TRG" ENABLE;
