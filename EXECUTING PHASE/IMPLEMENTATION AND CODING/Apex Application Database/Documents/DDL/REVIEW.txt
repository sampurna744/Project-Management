CREATE TABLE "REVIEW" 
   (	"REVIEW_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"REVIEW_DATE" DATE, 
	"REVIEW_SCORE" NUMBER(*,0), 
	"FEEDBACK" VARCHAR2(500), 
	"PRODUCT_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"USER_ID" NUMBER(*,0), 
	"REVIEW_PROCIDED" NUMBER, 
	"ORDER_ID" NUMBER NOT NULL ENABLE, 
	"CUSTOMER_ID" NUMBER NOT NULL ENABLE, 
	 CONSTRAINT "PK_REVIEW_ID" PRIMARY KEY ("REVIEW_ID")
  USING INDEX  ENABLE
   ) ;

  ALTER TABLE "REVIEW" ADD CONSTRAINT "FK_REVIEW_CUSTOMER_ID" FOREIGN KEY ("CUSTOMER_ID")
	  REFERENCES "CUSTOMER" ("CUSTOMER_ID") ON DELETE CASCADE ENABLE;
  ALTER TABLE "REVIEW" ADD CONSTRAINT "FK_REVIEW_PRODUCT_ID" FOREIGN KEY ("PRODUCT_ID")
	  REFERENCES "PRODUCT" ("PRODUCT_ID") ON DELETE CASCADE ENABLE;
  ALTER TABLE "REVIEW" ADD CONSTRAINT "FK_REVIEW_USER_ID" FOREIGN KEY ("USER_ID")
	  REFERENCES "CLECK_USER" ("USER_ID") ON DELETE CASCADE ENABLE;

  CREATE OR REPLACE EDITIONABLE TRIGGER "REVIEW_TRG" 
BEFORE INSERT ON REVIEW 
FOR EACH ROW 
BEGIN  
  IF :NEW.review_id IS NULL THEN  
  SELECT seq_review.NEXTVAL INTO :NEW.review_id FROM SYS.DUAL; 
  END IF; 
END;
/
ALTER TRIGGER "REVIEW_TRG" ENABLE;
