CREATE TABLE "CLECK_USER" 
   (	"USER_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"FIRST_NAME" VARCHAR2(100), 
	"LAST_NAME" VARCHAR2(100), 
	"USER_ADDRESS" VARCHAR2(100), 
	"USER_EMAIL" VARCHAR2(100), 
	"USER_AGE" NUMBER(*,0), 
	"USER_GENDER" CHAR(8), 
	"USER_PASSWORD" VARCHAR2(100), 
	"USER_PROFILE_PICTURE" VARCHAR2(255), 
	"USER_TYPE" VARCHAR2(10), 
	"USER_CONTACT_NO" NUMBER(*,0), 
	"USER_DOB" DATE, 
	 CONSTRAINT "PK_USER_ID" PRIMARY KEY ("USER_ID")
  USING INDEX  ENABLE
   ) ;

  CREATE OR REPLACE EDITIONABLE TRIGGER "USER_TRG" 
BEFORE INSERT ON Cleck_User 
FOR EACH ROW 
BEGIN  
  IF :NEW.user_id IS NULL THEN  
  SELECT seq_user.NEXTVAL INTO :NEW.user_id FROM SYS.DUAL; 
  END IF; 
END;
/
ALTER TRIGGER "USER_TRG" ENABLE;
