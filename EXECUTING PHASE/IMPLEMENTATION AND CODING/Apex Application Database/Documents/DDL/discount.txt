CREATE TABLE "DISCOUNT" 
   (	"DISCOUNT_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"DISCOUNT_OCCASSION" VARCHAR2(100), 
	"DISCOUNT_PERCENT" VARCHAR2(10), 
	"PRODUCT_ID" NUMBER(*,0) NOT NULL ENABLE, 
	 CONSTRAINT "PK_DISCOUNT_ID" PRIMARY KEY ("DISCOUNT_ID")
  USING INDEX  ENABLE
   ) ;

  ALTER TABLE "DISCOUNT" ADD CONSTRAINT "FK_PRODUCT_DISCOUNT_ID" FOREIGN KEY ("PRODUCT_ID")
	  REFERENCES "PRODUCT" ("PRODUCT_ID") ON DELETE CASCADE ENABLE;

  CREATE OR REPLACE EDITIONABLE TRIGGER "DISCOUNT_TRG" 
BEFORE INSERT ON DISCOUNT 
FOR EACH ROW 
BEGIN  
  IF :NEW.discount_id IS NULL THEN  
  SELECT seq_discount.NEXTVAL INTO :NEW.discount_id FROM SYS.DUAL; 
  END IF; 
END;
/
ALTER TRIGGER "DISCOUNT_TRG" ENABLE;
