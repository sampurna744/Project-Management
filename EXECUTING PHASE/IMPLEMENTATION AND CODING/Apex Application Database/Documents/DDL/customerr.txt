CREATE TABLE "CUSTOMER" 
   (	"CUSTOMER_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"CUSTOMER_DATE_JOINED" DATE, 
	"VERIFICATION_CODE" NUMBER(*,0), 
	"DATE_UPDATED" DATE, 
	"VERIFIED_CUSTOMER" NUMBER(1,0), 
	"USER_ID" NUMBER(*,0), 
	"PROFILE_PICTURE" VARCHAR2(255), 
	 CONSTRAINT "PK_CUSTOMER_ID" PRIMARY KEY ("CUSTOMER_ID")
  USING INDEX  ENABLE
   ) ;

  ALTER TABLE "CUSTOMER" ADD CONSTRAINT "FK_CUSTOMER_ID" FOREIGN KEY ("USER_ID")
	  REFERENCES "CLECK_USER" ("USER_ID") ON DELETE CASCADE ENABLE;

  CREATE OR REPLACE EDITIONABLE TRIGGER "CUSTOMER_TRG" 
BEFORE INSERT ON CUSTOMER 
FOR EACH ROW 
BEGIN  
  IF :NEW.customer_id IS NULL THEN  
  SELECT seq_customer.NEXTVAL INTO :NEW.customer_id FROM SYS.DUAL; 
  END IF; 
END;
/
ALTER TRIGGER "CUSTOMER_TRG" ENABLE;
