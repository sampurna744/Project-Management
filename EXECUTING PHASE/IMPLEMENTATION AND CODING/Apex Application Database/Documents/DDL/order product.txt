CREATE TABLE "ORDER_PRODUCT" 
   (	"ORDER_PRODUCT_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"NO_OF_PRODUCT" NUMBER, 
	"ORDER_STATUS" NUMBER, 
	"TOTAL_PRICE" NUMBER, 
	"SLOT_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"CART_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"ORDER_DATE" DATE, 
	"ORDER_TIME" TIMESTAMP (6), 
	"DISCOUNT_AMOUNT" NUMBER, 
	"CUSTOMER_ID" NUMBER, 
	 CONSTRAINT "PK_ORDER_PRODUCT_ID" PRIMARY KEY ("ORDER_PRODUCT_ID")
  USING INDEX  ENABLE
   ) ;

  ALTER TABLE "ORDER_PRODUCT" ADD CONSTRAINT "FK_CART_ID" FOREIGN KEY ("CART_ID")
	  REFERENCES "CART" ("CART_ID") ON DELETE CASCADE ENABLE;

  CREATE OR REPLACE EDITIONABLE TRIGGER "ORDER_TRG" 
BEFORE INSERT ON ORDER_PRODUCT 
FOR EACH ROW 
BEGIN  
  IF :NEW.order_product_id IS NULL THEN  
  SELECT seq_order.NEXTVAL INTO :NEW.order_product_id FROM SYS.DUAL; 
  END IF; 
END;
/
ALTER TRIGGER "ORDER_TRG" ENABLE;
